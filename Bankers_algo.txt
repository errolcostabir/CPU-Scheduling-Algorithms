int p[10];

int MAX[100][100];
int available[100];
int need[100][100];

void calculate_sequence(){
	int work[100];
	int flag = 0;
	bool finish[100];
	int k = 0;
	int sequence[100];
	for (i = 0;i < n;i++) {
		finish[i] = false;
	}
	for (i = 0;i < m;i++) {
		work[i] = available[i];
	}
	int g=0;
	for (int k = 0; k < n; k++) {
		for (i = 0; i < n; i++) {
			for (j = 0; j < m; j++) {
				if (finish[i] == false) {
					if (need[i][j] <= work[j]) {
						work[j] += allocation[i][j];
						finish[i] = true;
						sequence[g] = i;
						g++;
					}
					else
						break;
				}
			}
		}
	}
	cout << "the safe sequence is" << endl<<"<";
	for (i = 0;i < n;i++) {
		cout << sequence[i] << " ";
	}
	cout << ">";
}

void need_fun(){
	for (i = 0;i < n;i++) {
		for (j = 0;j < m;j++) {
			need[i][j] = MAX[i][j] - allocation[i][j];
		}
	}
	cout << "the NEED MAtrix is" << endl;
	for (i = 0;i < n;i++) {
		cout << "p[" << i << "]" << ": ";
		for (j = 0;j < m;j++) {
			cout<<need[i][j]<<" ";
		}
		cout << endl;
	}
	calculate_sequence();
}
